Ascending & Descending Sort Program Using Function

        
#include<stdio.h>

int ascending (int a[],int);
int descending (int a[],int);
int main()
{
	int i,j,n;
	printf("Enter the size of array");
	scanf("%d",&n);
	int a[n];
	for(i=0;i<n;i++)
	{
		printf("Enter the %d element",i+1);
		scanf("%d",&a[i]);
	}
	ascending(a,n);
	printf("The arrays in ascending order are :\n");
	for(i=0;i<n;i++)
	{
		printf("%d ",a[i]);
	}
	printf("\n");
	descending(a,n);
		printf("The arrays in descending order are :\n");
		for(i=0;i<n;i++)
	{
		printf("%d ",a[i]);
	}
	return 0;
}
int ascending(int a[],int n)
{
	int i,j,temp;
	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]>a[j])
			{
			temp=a[j];
			a[j]=a[i];
			a[i]=temp;	
			}
		}
	}
}
int descending(int a[],int n)
{
	int i,j,temp;
	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]<a[j])
			{
			temp=a[j];
			a[j]=a[i];
			a[i]=temp;	
			}
		}
	}
}

void add (int r,int c,int a[][c], int b[][c]);
int main()
{
	int r,c,i,j;
	printf("Enter the row  of  matrix ");
	scanf("%d",&r);
		printf("Enter the column matrix ");
	scanf("%d",&c);
	int a[r][c];
	int b[r][c];
	printf("Enter the element of first matrix \n");
	
	for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		{
			printf("Enter %d * %d element of matrix ",i+1,j+1);
			scanf("%d",&a[i][j]);
		}
	}
		printf("Enter the element of second matrix \n");
	for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		{
			printf("Enter %d * %d element of matrix ",i+1,j+1);
			scanf("%d",&b[i][j]);
		}
	}
	add(r,c,a,b);

	
}
void add (int r,int c, int a[][c],int b[][c] )
{
	int i,j;
	int d[r][c];
	for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		{
			d[i][j]=a[i][j]+b[i][j];
		}
	}
	
	printf("The sum of matrix is :\n");
		for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		{
			printf("%d   ",d[i][j]);
		}
		printf("\n");
	}
	
	
}
